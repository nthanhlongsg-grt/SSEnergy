# Backend — build qua SSH trên Mắt Bão (giống SGE baohanh.sgesolartech.vn)
# App chạy tại: ~/baohanh.ssenergy.vn/sse-api/
# KHÔNG dùng Application root = document root. Startup = dist/index.js (hoặc dist/start.cjs)

cd ~
export PATH="$HOME/.nodenv/versions/18/bin:$PATH"

# Xóa thư mục clone tạm, clone mới
rm -rf ~/SSEnergy-build
git clone https://github.com/nthanhlongsg-grt/SSEnergy.git SSEnergy-build

# Build
cd ~/SSEnergy-build/server
npm install
npm run build

# Tạo thư mục dist nếu chưa có (lần deploy đầu)
mkdir -p ~/baohanh.ssenergy.vn/sse-api/dist

# Copy dist mới vào app đang chạy (giữ nguyên .env và database trong sse-api/)
cp -r ~/SSEnergy-build/server/dist/* ~/baohanh.ssenergy.vn/sse-api/dist/

# Dọn dẹp
rm -rf ~/SSEnergy-build

# Restart Node (tìm process cũ)
ps aux | grep node
# kill -9 <PID>   hoặc Restart trong cPanel Node.js App

# Kiểm tra
curl -s http://127.0.0.1:3000/api/health
